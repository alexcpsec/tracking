apache_logs = readLines(apache_log_path)

dt_apache = data.table(timestamp = parsed_apache_ts, agent = parsed_apache_agent)

All right! Lets try to create a graph now!

dt_graph = rbind(data.table(Date=names(prop_verizon), Percentage=as.numeric(prop_verizon), ISP="Verizon"),
                 data.table(Date=names(prop_att), Percentage=as.numeric(prop_att), ISP="AT&T"))

ggplot(dt_graph, aes(Date, Percentage, fill=ISP)) +
    geom_bar(position="dodge", stat="identity") +
    theme(axis.text.x = element_text(angle = 45, hjust = 1, size=10)) +
    theme(axis.text.y = element_text(hjust = 1, size=12)) +
    ggtitle("Percentage of Verizon and AT&T UIDs over time\nAs seen on lessonslearned.org/sniff")
